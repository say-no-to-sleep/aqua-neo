const theme = document.documentElement.dataset.theme;
const slider = document.querySelector(".aqua-slider");

function syncValue(target) {
  const value = Number(target.dataset.value || 0);
  target.style.setProperty("--current-value", value);
}

slider?.addEventListener("pointerup", event => {
  syncValue(event.currentTarget);
});

Using Aqua2
